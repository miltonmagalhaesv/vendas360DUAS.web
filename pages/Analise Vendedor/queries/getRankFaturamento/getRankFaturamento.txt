SELECT
  vendedor,                                        -- ou nome_vendedor se jรก padronizado
  ROUND(SUM(valor_liquido)::numeric, 2) AS faturamento_liquido
FROM public.vw_vendedor_mes
WHERE ano = {{ Number(selAno.selectedOptionValue) }}
GROUP BY 1
ORDER BY faturamento_liquido DESC
LIMIT 15;
