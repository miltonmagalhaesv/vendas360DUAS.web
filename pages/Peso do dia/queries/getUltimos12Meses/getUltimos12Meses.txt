WITH limites AS (
  SELECT
    DATE '2025-01-01' AS inicio,
    date_trunc(
      'month',
      make_date((extract(year FROM current_date)::int) + 1, 5, 1)  -- Maio do próximo ano
    )::date AS fim
)
SELECT 
  to_char(mes, 'YYYY-MM') AS value,
  CASE extract(month from mes)
    WHEN 1 THEN 'Janeiro'
    WHEN 2 THEN 'Fevereiro'
    WHEN 3 THEN 'Março'
    WHEN 4 THEN 'Abril'
    WHEN 5 THEN 'Maio'
    WHEN 6 THEN 'Junho'
    WHEN 7 THEN 'Julho'
    WHEN 8 THEN 'Agosto'
    WHEN 9 THEN 'Setembro'
    WHEN 10 THEN 'Outubro'
    WHEN 11 THEN 'Novembro'
    WHEN 12 THEN 'Dezembro'
  END || '/ ' || extract(year from mes) AS label
FROM (
  SELECT generate_series(inicio, fim, interval '1 month')::date AS mes
  FROM limites
) meses
ORDER BY mes;
